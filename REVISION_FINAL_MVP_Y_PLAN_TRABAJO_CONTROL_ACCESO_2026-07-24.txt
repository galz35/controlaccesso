12. MIGRACIÓN MVP PROPUESTA
======================================================================

Crear:

  database/008_mvp_stabilization.sql

Contenido mínimo:

  USE ControlAcceso;
  GO

  CREATE OR ALTER PROCEDURE
  dbo.sp_Acceso_RegistrarSalida
    @Id INT,
    @EdificioIdAutorizado INT = NULL
  AS
  BEGIN
    SET NOCOUNT ON;

    UPDATE dbo.tblRegistroAcceso
    SET FechaSalida = GETDATE()
    WHERE Id = @Id
      AND FechaSalida IS NULL
      AND (
        @EdificioIdAutorizado IS NULL OR
        EdificioId = @EdificioIdAutorizado
      );

    IF @@ROWCOUNT = 0
      THROW 51000,
        'Registro no encontrado, cerrado o no autorizado.',
        1;

    SELECT
      Id,
      NombrePersona AS Nombre,
